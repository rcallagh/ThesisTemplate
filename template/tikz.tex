\newcommand\GofL[1]{
  \foreach[count=\yi] \y in #1 {
    \foreach[count=\xi] \x in \y {%
      \edef\one{1}
      \ifx\x\one
      \node[cellframe,alivecell] at (\xi, -\yi) {};
      \else
      \node[cellframe] at (\xi, -\yi) {};
      \fi
    }
  }
}
