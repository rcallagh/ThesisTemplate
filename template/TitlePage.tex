\def\maketitle{%
  \newbox\@crestbox
  
  %% Title page
  \begin{titlepage}
  	\begin{center}
    \includegraphics[width=0.5\columnwidth]{template/logos/UCL_logo.eps}
    \vfill%

    % \vspace*{20mm}%
    {\def\baselinestretch{1.2}\Huge\bfseries \@title \par} % Title
    \vskip 25mm%

    {\huge\bfseries \@author} % Author
    \vskip 10mm%


    {\Large \@provisionaltext} % Provisional text
    \vfill

    {\large
      \begin{description}
        \centering
        \@supervisorstext
      \end{description}
    } % Supervisor
    \vskip 5mm

    {\large A thesis submitted for the degree of}\vskip 4mm
    {\large \bfseries \@degree } \vskip 2mm
    {\large in } \vskip 2mm
    {\large \bfseries \@subject} \vskip 4mm
    {\large as part of} \vskip 2mm
    {\large \bfseries \@school } \vskip 2mm
    {\large \bfseries \@program} % Program
    \vskip 30mm
    \vfill

    \@thesisdate % Date\
  \end{center}
\end{titlepage}
\funding
 
}
